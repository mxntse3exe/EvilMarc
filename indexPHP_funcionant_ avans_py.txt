                            $tamanoMaximo = 650 * 1024 * 1024; // 650 MB en bytes

                            // Verificar si el directorio tiene permisos de escritura
                            if (!is_writable($directorio)) {
                                echo "<p style='color: red;'>Error: el directori no té permisos d'escriptura.</p>";
                            } else {
                                // Verificar si el archivo supera el tamaño máximo permitido
                                if ($_FILES['fileToUpload']['size'] > $tamanoMaximo) {
                                    echo "<p style='color: red;'>L'arxiu seleccionat és massa pesat. El límit és de 650 MB.</p>";
                                } else {
                                    // Intentar mover el archivo al directorio de destino
                                    if (move_uploaded_file($_FILES['fileToUpload']['tmp_name'], rtrim($directorio, '/') . '/' . $_FILES['fileToUpload']['name'])) {

                                    } else {
